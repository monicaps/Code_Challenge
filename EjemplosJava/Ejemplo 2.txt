import java.util.*;

class Objeto {
    char naturaleza;
    double importe;
    Date fecha;
    // ... otros atributos y constructores
}

public class AgrupacionObjetos {
    public static Map<Double, List<Objeto>> agruparPorImporte(List<Objeto> lista) {
        Map<Double, List<Objeto>> mapeo = new HashMap<>();

        for (int i = 0; i < lista.size() - 1; i++) {
            Objeto obj1 = lista.get(i);
            Objeto obj2 = lista.get(i + 1);

            // Verificar si los objetos tienen la misma naturaleza (si es necesario)
            if (obj1.naturaleza == obj2.naturaleza) {
                double suma = obj1.importe + obj2.importe;
                mapeo.computeIfAbsent(suma, k -> new ArrayList<>()).addAll(Arrays.asList(obj1, obj2));
            }
        }

        return mapeo;
    }
}
